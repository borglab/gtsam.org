105 Vector evaluateError(const Rot3& iRc, boost::optional<Matrix&> H = boost::none) const override {
106 Unit3 i_q = iRc * c_z_;
107 Vector error = i_p_.error(i_q, H);
108 if (H) {

110 iRc.rotate(c_z_, DR);
