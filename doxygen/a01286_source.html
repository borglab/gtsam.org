58 DecisionTreeFactor toDecisionTreeFactor() const override {

62 std::vector<double> table;
63 for (size_t i1 = 0; i1 < cardinality0_; i1++)
64 for (size_t i2 = 0; i2 < cardinality1_; i2++) table.push_back(i1 != i2);
